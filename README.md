```python
def create_mat(a):   
    mats = []
    for line in a:
        mat = []
        for i in range(len(line)):
            mat.append(np.roll(line, i))
        mats.append(np.array(mat))
    return mats
```


```python
def around(obj):
    if obj.shape[0] == 1:
        return obj
    top = obj[0,0:-1]
    right = obj[0:-1,-1]
    bottom = np.flip(obj[-1,1:])
    left = np.flip(obj[1:,0])
    line = np.concatenate([top, right, bottom, left])
    return line

def spin(obj):
    lines = []
    m = obj
    for i in range(obj.shape[0]//2 + 1):
        lines.append(around(m))
        m = m[1:-1,1:-1]
    return lines
```
